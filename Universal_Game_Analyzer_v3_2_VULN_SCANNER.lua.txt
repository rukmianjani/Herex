local function SafeCall(func, ...)
    local success, result = pcall(func, ...)
    return success, result
end

local function GetPath(obj)
    local success, path = SafeCall(function() return obj:GetFullName() end)
    return success and path or "Unknown"
end

local function CountObjects(container)
    local count = 0
    SafeCall(function()
        for _ in pairs(container:GetDescendants()) do count = count + 1 end
    end)
    return count
end

local function IsClientEditable(obj)
    if not obj:IsA("ValueBase") then return false end
    local originalValue = obj.Value
    local success = false
    SafeCall(function()
        local testValue
        if obj:IsA("NumberValue") or obj:IsA("IntValue") then
            testValue = originalValue + 1
        elseif obj:IsA("BoolValue") then
            testValue = not originalValue
        else
            testValue = originalValue
        end
        obj.Value = testValue
        wait(0.05)
        if obj.Value == testValue then success = true end
        obj.Value = originalValue
    end)
    return success
end

function Analyzer:ScanVulnerabilities()
    print("\nğŸ”“ SCANNING FOR VULNERABILITIES...")
    
    self.Vulnerabilities = {Critical = {}, High = {}, Medium = {}, Low = {}, Info = {}}
    local totalVulns = 0
    
    -- Scan all scripts
    for _, scriptData in ipairs(self.DeepScan.Scripts) do
        local scriptSource = ""
        
        -- Try to get script source
        SafeCall(function()
            if decompile then
                scriptSource = decompile(scriptData.Object)
            elseif scriptData.Object:IsA("ModuleScript") then
                pcall(function()
                    local req = require(scriptData.Object)
                    if type(req) == "table" then
                        scriptSource = Services.HttpService:JSONEncode(req)
                    end
                end)
            end
        end)
        
        local contentToScan = scriptSource .. " " .. scriptData.Name .. " " .. scriptData.Path
        
        if #contentToScan > 5 then
            for severity, patterns in pairs(VulnPatterns) do
                for _, vuln in ipairs(patterns) do
                    if contentToScan:match(vuln.pattern) then
                        table.insert(self.Vulnerabilities[severity], {
                            Name = vuln.name,
                            Description = vuln.desc,
                            ExploitMethod = vuln.exploit,
                            DataToModify = vuln.data,
                            ScriptName = scriptData.Name,
                            ScriptPath = scriptData.Path,
                            ScriptType = scriptData.Type,
                            Severity = severity,
                            Code = self:GenerateVulnExploit(vuln, scriptData)
                        })
                        totalVulns = totalVulns + 1
                    end
                end
            end
        end
    end
    
    -- Scan remotes for vulnerable names
    local remoteVulns = {
        {p="money", s="Critical"}, {p="cash", s="Critical"}, {p="coin", s="Critical"},
        {p="gold", s="Critical"}, {p="gem", s="Critical"}, {p="give", s="Critical"},
        {p="admin", s="Critical"}, {p="buy", s="High"}, {p="sell", s="High"},
        {p="add", s="High"}, {p="set", s="High"}, {p="level", s="High"},
        {p="xp", s="High"}, {p="damage", s="High"}, {p="speed", s="High"},
        {p="teleport", s="High"}, {p="spawn", s="Medium"}, {p="claim", s="Medium"},
    }
    
    for _, remote in ipairs(self.DeepScan.RemoteCommunication) do
        local name = remote.Name:lower()
        for _, rv in ipairs(remoteVulns) do
            if name:match(rv.p) then
                table.insert(self.Vulnerabilities[rv.s], {
                    Name = rv.p:upper() .. " Remote",
                    Description = "Remote terkait " .. rv.p,
                    ExploitMethod = "Fire dengan parameter manipulasi",

                    ScriptName = remote.Name,
                    ScriptPath = remote.Path,
                    ScriptType = remote.Type,
                    Severity = rv.s,
                    Code = self:GenerateRemoteVulnCode(remote)
                })
                totalVulns = totalVulns + 1

        end
    end
    
    print("ğŸ”“ Found " .. totalVulns .. " vulnerabilities!")

end

        self.ScanData.Scanned = 0
        
        self.Exploits = {Local = {}, Remote = {}, Automation = {}}

function Analyzer:ScanContainer(container)

    end)
end
