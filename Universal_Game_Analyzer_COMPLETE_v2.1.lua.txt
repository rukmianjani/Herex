local function SafeCall(func, ...)
    local success, result = pcall(func, ...)
    return success, result
end

local function GetPath(obj)
    local success, path = SafeCall(function() return obj:GetFullName() end)
    return success and path or "Unknown"
end

local function CountObjects(container)
    local count = 0
    SafeCall(function()
        for _, obj in pairs(container:GetDescendants()) do
            count = count + 1
        end
    end)
    return count
end

local function IsExploitable(obj)
    if obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction") then return true end
    if obj:FindFirstChildOfClass("ClickDetector") then return true end
    if obj:FindFirstChildOfClass("ProximityPrompt") then return true end
    if obj:IsA("Tool") then return true end
    if obj:IsA("IntValue") or obj:IsA("NumberValue") then
        local name = obj.Name:lower()
        if name:match("money") or name:match("cash") or name:match("coin") or 
           name:match("gold") or name:match("level") then
            return true
        end
    end
    return false
end

local function DetectRemoteParameters(remoteName, remotePath)
    local params = {}
    local name = remoteName:lower()
    
    -- Detect common parameter patterns
    if name:match("buy") or name:match("purchase") then
        table.insert(params, {name = "ItemName/ID", type = "string", default = "Item1", suggestions = {"Item1", "Gear1", "Platform1"}})
        table.insert(params, {name = "Amount", type = "number", default = 1, suggestions = {1, 10, 100}})
        table.insert(params, {name = "Price", type = "number", default = 0, suggestions = {0, 100, 1000}})
    elseif name:match("sell") then
        table.insert(params, {name = "ItemName/ID", type = "string", default = "All", suggestions = {"All", "Item1"}})
        table.insert(params, {name = "Amount", type = "number", default = 1, suggestions = {1, 999}})
    elseif name:match("claim") or name:match("reward") then
        table.insert(params, {name = "RewardID", type = "string", default = "Daily", suggestions = {"Daily", "Quest1"}})
    elseif name:match("upgrade") then
        table.insert(params, {name = "UpgradeType", type = "string", default = "Speed", suggestions = {"Speed", "Damage"}})
        table.insert(params, {name = "Level", type = "number", default = 1, suggestions = {1, 5, 10}})
    elseif name:match("equip") then
        table.insert(params, {name = "ItemID", type = "string", default = "Tool1", suggestions = {"Tool1", "Weapon1"}})
    elseif name:match("gift") or name:match("pass") then
        table.insert(params, {name = "PassID", type = "number", default = 1, suggestions = {1, 2, 3}})
        table.insert(params, {name = "RecipientID", type = "number", default = player.UserId, suggestions = {player.UserId}})
    end
    
    -- Always add some common test combinations
    if #params == 0 then
        table.insert(params, {name = "Arg1", type = "auto", default = nil, suggestions = {nil, true, 1, "test"}})
    end
    
    return params
end

local function GenerateTestArguments(params)
    -- Generate multiple test combinations
    local combinations = {
        {}, -- Empty
        {nil}, -- Single nil
        {true}, -- Boolean
        {1}, -- Number
        {"test"}, -- String
    }
    
    -- Generate from detected params
    if #params > 0 then
        for _, param in ipairs(params) do
            for _, suggestion in ipairs(param.suggestions) do
                table.insert(combinations, {suggestion})
            end
        end
        
        -- Multi-param combinations
        if #params >= 2 then
            table.insert(combinations, {params[1].default, params[2].default})
            table.insert(combinations, {params[1].suggestions[1], params[2].suggestions[1]})
        end
    end
    
    return combinations
end

function Analyzer:StartScan()
    if self.IsScanning then
        self:Notify("âš ï¸ Warning", "Already scanning!")
        return
    end
    
    self.IsScanning = true
    self:ResetData()
    self.ScanData.StartTime = tick()
    
    self:Notify("ğŸ” Scanner", "Starting deep scan...")
    
    spawn(function()
        print("\nğŸ“Š Counting objects...")
        local containers = {
            {obj = Services.WS, name = "Workspace"},
            {obj = Services.RS, name = "ReplicatedStorage"},
            {obj = playerGui, name = "PlayerGui"}
        }
        
        for _, container in ipairs(containers) do
            self.ScanData.Total = self.ScanData.Total + CountObjects(container.obj)
        end
        
        print("ğŸ“Š Total to scan: " .. self.ScanData.Total)
        
        if GUI_UpdateProgress then
            GUI_UpdateProgress(0, self.ScanData.Total)
        end
        
        for _, container in ipairs(containers) do
            print("\nğŸ” Scanning " .. container.name .. "...")
            self:ScanContainer(container.obj, container.name)
        end
        
        self.ScanData.Duration = tick() - self.ScanData.StartTime
        self.IsScanning = false
        
        self:AnalyzeFeatures()
        self:GenerateScripts()
        
        print("\nâœ… Scan complete!")
        print("â±ï¸ Duration: " .. math.floor(self.ScanData.Duration) .. "s")
        print("ğŸ“Š Objects scanned: " .. self.ScanData.Scanned)
        print("ğŸ¯ Exploits found: " .. self:GetTotalExploits())
        print("ğŸ“œ Scripts generated: " .. #self.Scripts)
        
        self:Notify("âœ… Complete!", string.format("Found %d exploits in %ds", self:GetTotalExploits(), math.floor(self.ScanData.Duration)))
        
        if GUI_UpdateResults then
            GUI_UpdateResults()
        end
    end)
end

function Analyzer:ScanContainer(container, name)
    SafeCall(function()
        local descendants = container:GetDescendants()
        
        for _, obj in ipairs(descendants) do
            self.ScanData.Scanned = self.ScanData.Scanned + 1
            
            if self.ScanData.Scanned % 100 == 0 then
                if GUI_UpdateProgress then
                    GUI_UpdateProgress(self.ScanData.Scanned, self.ScanData.Total)
                end
                wait()
            end
            
            if IsExploitable(obj) then
                self:CatalogFeature(obj)
            end
        end
    end)
end

function Analyzer:CatalogFeature(obj)
    SafeCall(function()
        local objType = obj.ClassName
        local objName = obj.Name
        local objPath = GetPath(obj)
        
        if objType == "RemoteEvent" or objType == "RemoteFunction" then
            table.insert(self.Features.Remotes, {
                Type = objType,
                Name = objName,
                Path = objPath,
                Object = obj
            })
        end
        
        if obj:FindFirstChildOfClass("ClickDetector") then
            table.insert(self.Features.Clickables, {
                Name = objName,
                Path = objPath,
                Object = obj,
                Detector = obj:FindFirstChildOfClass("ClickDetector")
            })
        end
        
        if obj:FindFirstChildOfClass("ProximityPrompt") then
            table.insert(self.Features.Prompts, {
                Name = objName,
                Path = objPath,
                Object = obj,
                Prompt = obj:FindFirstChildOfClass("ProximityPrompt")
            })
        end
        
        if objType == "Tool" then
            table.insert(self.Features.Tools, {
                Name = objName,
                Path = objPath,
                Object = obj
            })
        end
        
        if objType:match("Value") then
            local nameLower = objName:lower()
            if nameLower:match("money") or nameLower:match("cash") or 
               nameLower:match("coin") or nameLower:match("gold") or 
               nameLower:match("level") or nameLower:match("xp") then
                table.insert(self.Features.Values, {
                    Type = objType,
                    Name = objName,
                    Path = objPath,
                    Object = obj,
                    Value = obj.Value
                })
            end
        end
    end)
end

function Analyzer:GenerateRemoteCode(exploit, testArgs)
    local code = string.format([[
-- Remote Executor: %s
-- Type: %s
-- Path: %s

local remote = nil

-- Find remote by path
for _, obj in pairs(game:GetDescendants()) do
    if obj:GetFullName() == "%s" then
        remote = obj
        break
    end
end

if not remote then
    warn("âŒ Remote not found: %s")
    return
end

print("âœ… Found remote:", remote:GetFullName())
print("Type:", remote.ClassName)

]], exploit.Name, exploit.RemoteType, exploit.Path, exploit.Path, exploit.Name)

    -- Add smart execution with multiple test arguments
    code = code .. [[
-- Test multiple argument combinations
local testCombinations = {
]]

    for i, args in ipairs(testArgs) do
        local argsStr = "{"
        for j, arg in ipairs(args) do
            if type(arg) == "string" then
                argsStr = argsStr .. '"' .. arg .. '"'
            elseif type(arg) == "nil" then
                argsStr = argsStr .. "nil"
            else
                argsStr = argsStr .. tostring(arg)
            end
            if j < #args then argsStr = argsStr .. ", " end
        end
        argsStr = argsStr .. "}"
        code = code .. "    " .. argsStr .. ",\n"
    end

    code = code .. [[}

local successCount = 0
local failCount = 0

for i, args in ipairs(testCombinations) do
    local success, result = pcall(function()
        if remote:IsA("RemoteEvent") then
            remote:FireServer(unpack(args))
        else
            return remote:InvokeServer(unpack(args))
        end
    end)
    
    if success then
        successCount = successCount + 1
        print(string.format("âœ… Test %d SUCCESS with args:", i), unpack(args))
        if result then
            print("  Result:", result)
        end
    else
        failCount = failCount + 1
        print(string.format("âŒ Test %d FAILED:", i), result)
    end
    
    wait(0.5)
end

print("\nğŸ“Š Results:")
print("Success:", successCount)
print("Failed:", failCount)
]]

    return code
end

function Analyzer:ExecuteScript(script, customArgs)
    if not script or not script.Code then
        self:Notify("âŒ Error", "Invalid script!")
        return
    end
    
    self:Notify("â–¶ï¸ Execute", "Running: " .. script.Name)
    
    spawn(function()
        local codeToRun = script.Code
        
        -- If custom args provided, modify the code
        if customArgs and #customArgs > 0 then
            print("ğŸ”§ Using custom arguments:", table.concat(customArgs, ", "))
        end
        
        local success, err = pcall(function()
            loadstring(codeToRun)()
        end)
        
        if success then
            print("âœ… Executed: " .. script.Name)
            self:Notify("âœ… Success", script.Name .. " executed!")
            
            table.insert(self.Active, {
                Name = script.Name,
                Time = tick()
            })
        else
            print("âŒ Failed: " .. script.Name)
            print("Error: " .. tostring(err))
            self:Notify("âŒ Failed", "Check console for details")
        end
    end)
end

            if script.Exploit == exploit then

                wait(1)
            end
        end
    end
end

    self:Notify("â¹ï¸ Stopped", "All executions stopped")
end

function GUI_UpdateExploitTabs()
    -- Clear all exploit tabs
    for _, tabName in ipairs({"High", "Medium", "Low", "All"}) do
        local content = tabContents[tabName]
        for _, child in ipairs(content:GetChildren()) do
            if child:IsA("Frame") or child:IsA("TextLabel") then
                child:Destroy()
            end
        end
    end
    
    -- Update High Priority Tab
    local highContent = tabContents["High"]
    CreateSection(highContent, "ğŸ”¥ HIGH PRIORITY EXPLOITS (" .. #Analyzer.Priority.High .. ")")
    
    if #Analyzer.Priority.High == 0 then
        CreateLabel(highContent, "No high priority exploits found", 50)
    else
        for _, exploit in ipairs(Analyzer.Priority.High) do
            CreateExploitCard(highContent, exploit)
        end
    end
    
    -- Update Medium Priority Tab
    local mediumContent = tabContents["Medium"]
    CreateSection(mediumContent, "âš¡ MEDIUM PRIORITY EXPLOITS (" .. #Analyzer.Priority.Medium .. ")")
    
    if #Analyzer.Priority.Medium == 0 then
        CreateLabel(mediumContent, "No medium priority exploits found", 50)
    else
        for _, exploit in ipairs(Analyzer.Priority.Medium) do
            CreateExploitCard(mediumContent, exploit)
        end
    end
    
    -- Update Low Priority Tab
    local lowContent = tabContents["Low"]
    CreateSection(lowContent, "ğŸ“Œ LOW PRIORITY EXPLOITS (" .. #Analyzer.Priority.Low .. ")")
    
    if #Analyzer.Priority.Low == 0 then
        CreateLabel(lowContent, "No low priority exploits found", 50)
    else
        for _, exploit in ipairs(Analyzer.Priority.Low) do
            CreateExploitCard(lowContent, exploit)
        end
    end
    
    -- Update All Tab
    local allContent = tabContents["All"]
    CreateSection(allContent, "ğŸ¯ ALL EXPLOITS (" .. Analyzer:GetTotalExploits() .. ")")
    
    local allExploits = {}
    for _, priority in pairs(Analyzer.Priority) do
        for _, exploit in ipairs(priority) do
            table.insert(allExploits, exploit)
        end
    end
    
    if #allExploits == 0 then
        CreateLabel(allContent, "No exploits found yet. Run a scan!", 50)
    else
        for _, exploit in ipairs(allExploits) do
            CreateExploitCard(allContent, exploit)
        end
    end
end
